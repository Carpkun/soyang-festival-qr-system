# GitHub 저장소 연결 명령어
# GitHub에서 저장소 생성 후 아래 명령어들을 순서대로 실행하세요

# 1. 원격 저장소 추가 (your-username을 실제 GitHub 사용자명으로 변경)
git remote add origin https://github.com/your-username/soyang-festival-qr-system.git

# 2. 기본 브랜치 이름을 main으로 설정
git branch -M main

# 3. GitHub에 푸시
git push -u origin main

# 확인: 원격 저장소 연결 상태 확인
git remote -v
